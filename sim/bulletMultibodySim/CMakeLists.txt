        "${CMAKE_CXX_FLAGS_DEBUG} -D_DEBUG -Wl,--no-as-needed -O0 -mtune=native -march=native")

        "-Wl,--no-as-needed -O3 -mtune=native -march=native")

        )

        BulletRobotics
